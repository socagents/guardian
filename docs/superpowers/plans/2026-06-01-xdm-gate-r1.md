### Task 1: Reverse-engineering helper — extract each source's gate from its live rules

**Files:**
- Create: `scripts/maintainer/reverse_engineer_gate.py`
- Read-only inputs: `scripts/maintainer/modeling_rules/<Pack>__<Rule>.xif`, `scripts/maintainer/parsing_rules/<Pack>__<Rule>.xif`

This is analysis tooling — it READS the live rules each run (not a stored mapping). It prints, per source id, the gate field+value(s), its kind (`raw` | `computed` | `meta`), and for computed gates the underlying `alter <field> = <expr>` it resolves from.

- [ ] **Step 1: Write the script**

```python
#!/usr/bin/env python3
"""Reverse-engineer a data source's modeling-rule gate from its .xif.
Reads the live rule each run — no stored mapping. Usage:
  python3 scripts/maintainer/reverse_engineer_gate.py <data_source_id> [<id> ...]
"""
import os, re, sys, glob
MR = os.path.join(os.path.dirname(__file__), "modeling_rules")
PR = os.path.join(os.path.dirname(__file__), "parsing_rules")

def _xif_for(pack, rule, root):
    p = os.path.join(root, f"{pack}__{rule}.xif")
    if os.path.exists(p): return p
    for x in glob.glob(os.path.join(root, "*.xif")):
        if os.path.basename(x).startswith(pack + "__"): return x
    return None

def _model_block(txt, dataset):
    m = re.search(r"\[MODEL:\s*dataset\s*=\s*" + re.escape(dataset) + r"\s*\](.*?)(?=\[MODEL:|\Z)", txt, re.S)
    return m.group(1) if m else txt

def analyze(ds_id):
    pack, rule, dataset = ds_id.split("__", 2)
    mr = _xif_for(pack, rule, MR)
    if not mr:
        return {"id": ds_id, "error": "no modeling rule .xif found"}
    txt = open(mr, errors="ignore").read()
    blk = _model_block(txt, dataset)
    fm = re.search(r"filter\s+([^\n|]+)", blk)
    gate = fm.group(1).strip() if fm else None
    field = re.match(r"\s*([A-Za-z0-9_.]+)", gate).group(1) if gate else None
    kind = "meta" if field and field.startswith("_") else "raw"
    # computed: the gate field is assigned earlier via `alter <field> = ...`
    computed_expr = None
    if field:
        am = re.search(r"alter[^\n]*\b" + re.escape(field) + r"\s*=\s*([^\n,]+)", blk)
        if am:
            kind, computed_expr = "computed", am.group(1).strip()
    return {"id": ds_id, "dataset": dataset, "modeling_rule": os.path.basename(mr),
            "gate": gate, "gate_field": field, "kind": kind, "computed_expr": computed_expr}

if __name__ == "__main__":
    import json
    for ds in sys.argv[1:]:
        print(json.dumps(analyze(ds), indent=2))
```
